Sudoku Solver

Sudoku Puzzle Solver, using the backtracking algorithm.

Project Summary

This Sudoku Solver allows for the input of a 9X9 grid (in the form of a list) and computates a solution using the backrtacking algorithm recursively. This sudoku solver was my first experience with the backtracking algorithm and it made me alot more knowledgeable into the power of it.

Developed with: Python

Source Code



	import random

	'''
	Sudoku Board Solver, takes valid sudoku board, and using backtrack - solves it.
	Developed by Noah Smiley
	'''
	board = [[7,8,0,4,0,0,1,2,0],
			[6,0,0,0,7,5,0,0,9],
			[0,0,0,6,0,1,0,7,8],
			[0,0,7,0,4,0,2,6,0],
			[0,0,1,0,5,0,9,3,0],
			[9,0,4,0,6,0,0,0,5],
			[0,7,0,3,0,0,0,1,2],
			[1,2,0,0,0,7,4,0,0],
			[0,4,9,2,0,6,0,0,7]]
	def solve(bo):
		find = find_empty(bo)
		if not find:
			return True
		else:
			row, col = find
		for i in range(1,10):
			if valid(bo, i, (row, col)):
				bo[row][col] = i
				if solve(bo):
					return True
				bo[row][col] = 0
		return False
	def valid(bo, num, pos):
		# Check row
		for i in range(len(bo[0])):
			if bo[pos[0]][i] == num and pos[1] != i:
				return False
		# Check column
		for i in range(len(bo)):
			if bo[i][pos[1]] == num and pos[0] != i:
				return False
		# Check box
		box_x = pos[1] // 3
		box_y = pos[0] // 3
		for i in range(box_y*3, box_y*3 + 3):
			for j in range(box_x * 3, box_x*3 + 3):
				if bo[i][j] == num and (i,j) != pos:
					return False
		return True
	def print_board(bo):
		i=0
		j=0
		for i in range(len(bo[i])):
			if i%3 == 0 and i != 0:
				print('- - - - - - - - - - - - - - - - - - - - - - - ')
			for j in range(len(bo[i])):
				if j % 3 == 0 and j !=0:
					print('|',end = '')
				print('  '+str(bo[i][j])+'  ',end = '')  
				j+=1
			print('\n')    
			i+=1
	def find_empty(bo):
		for i in range(len(bo)):
			for j in range(len(bo[0])):
				if bo[i][j] == 0:
					return (i, j)  # row, col	
		return None
	print_board(board)
	print( "\n---------------------------------------------------\n")
	solve(board)
	print_board(board)
